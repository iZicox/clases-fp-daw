cmd de push

# 1️⃣ Ver el estado actual del repositorio
git status

# 2️⃣ Agregar los archivos nuevos o modificados al área de preparación (staging)
git add .

# 3️⃣ Crear un commit con un mensaje descriptivo
git commit -m "Descripción del cambio que hiciste"

# 4️⃣ (Opcional) Ver la rama actual
git branch

# 5️⃣ Subir (push) los cambios a la rama principal del repositorio remoto
git push origin main
